<h2 id="list-bank-accounts">List bank accounts</h2><pre class=prettyprint><code class=>GET https://api.freeagent.com/v2/bank_accounts
</code></pre><h3>Input</h3><h4>View Filters</h4><pre class=prettyprint><code class=>GET https://api.freeagent.com/v2/bank_accounts?view=standard_bank_accounts
</code></pre>
<ul>
<li><code>standard_bank_accounts</code>: Show only standard bank accounts.</li>
<li><code>credit_card_accounts</code>: Show only credit card accounts.</li>
<li><code>paypal_accounts</code>: Show only paypal accounts.</li>
</ul>
