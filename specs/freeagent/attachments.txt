          <h1>Attachments</h1>
<p><em>Minimum access level</em>: To access attachments you must have the same access level as the item the file is attached to.</p>
<h2 id="attributes">Attributes</h2>
<table>
  <thead>
    <tr>
      <th>Attribute</th>
      <th>Description</th>
      <th>Kind</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>url</td>
      <td>The unique identifier for the attachment</td>
      <td>URI</td>
    </tr>
    <tr>
      <td>content_src</td>
      <td>URL to the original full-size file</td>
      <td>URI</td>
    </tr>
    <tr>
      <td>content_src_medium</td>
      <td>URL to a medium-size thumbnail (if a thumbnail exists)</td>
      <td>URI</td>
    </tr>
    <tr>
      <td>content_src_small</td>
      <td>URL to a small-size thumbnail (if a thumbnail exists)</td>
      <td>URI</td>
    </tr>
    <tr>
      <td>expires_at</td>
      <td>When the <code>content_src</code> URLs expire. New URLs should be fetched from the API once this timestamp passes.</td>
      <td>Timestamp</td>
    </tr>
    <tr>
      <td>content_type</td>
      <td>MIME-type</td>
      <td>String</td>
    </tr>
    <tr>
      <td>file_name</td>
      <td>Original name of the file</td>
      <td>String</td>
    </tr>
    <tr>
      <td>file_size</td>
      <td>Size of original file in bytes</td>
      <td>Integer</td>
    </tr>
    <tr>
      <td>description</td>
      <td>Free-text description</td>
      <td>String</td>
    </tr>
  </tbody>
</table>
